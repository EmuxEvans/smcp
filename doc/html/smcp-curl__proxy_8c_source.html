76 static smcp_status_t
77 resend_async_response(void* context) {
78  smcp_status_t ret = 0;
79  smcp_curl_request_t request = (smcp_curl_request_t)context;
80  struct smcp_async_response_s* async_response = &request->async_response;
81 
82  size_t len = request->content_len;
83  if(len>512) {
84  len = 512;
85  }
86 
87  {
88  long code;
89  curl_easy_getinfo(request->curl, CURLINFO_RESPONSE_CODE,&code);
90  code = http_to_coap_code(code);
91 
92  ret = smcp_outbound_begin_response(code);
93  require_noerr(ret,bail);
94  }
95 
96  ret = smcp_outbound_set_async_response(async_response);
97  require_noerr(ret,bail);
98 
99  {
100  const char* content_type_string;
101  curl_easy_getinfo(request->curl, CURLINFO_CONTENT_TYPE,&content_type_string);
102  coap_content_type_t content_type = coap_content_type_from_cstr(content_type_string);
103  if(content_type!=COAP_CONTENT_TYPE_UNKNOWN) {
104  ret = smcp_outbound_add_option_uint(COAP_OPTION_CONTENT_TYPE, content_type);
105  check_noerr(ret);
106  } else {
107  DEBUG_PRINTF("Unrecognised content-type: %s",content_type_string);
108  }
109  }
110 
111  ret = smcp_outbound_append_content(request->content, len);
112  require_noerr(ret,bail);
113 
114  ret = smcp_outbound_send();
115  require_noerr(ret,bail);
116 
117  if(ret) {
118  assert_printf(
119  "smcp_outbound_send() returned error %d(%s).\n",
120  ret,
121  smcp_status_to_cstr(ret)
122  );
123  goto bail;
124  }
125 
126 bail:
127  return ret;
128 }

149 static size_t
150 WriteMemoryCallback(void *contents, size_t size, size_t nmemb, void *userp)
151 {
152  size_t realsize = size * nmemb;
153  smcp_curl_request_t request = (smcp_curl_request_t)userp;
154 
155  require_action(realsize>=0,bail,realsize=0);
156 
157  // At the moment limit to one call.
158  require_action(request->content==NULL,bail,realsize=0);
159 
160  request->content = realloc(request->content, request->content_len + realsize + 1);
161 
162  require_action(request->content!=NULL,bail,realsize=0);
163 
164  memcpy(&(request->content[request->content_len]), contents, realsize);
165  request->content_len += realsize;
166  request->content[request->content_len] = 0;
167 
168  smcp_transaction_init(
169  &request->async_transaction,
170  0, // Flags
171  (void*)&resend_async_response,
172  (void*)&async_response_ack_handler,
173  (void*)request
174  );
175 
176  smcp_transaction_begin(
177  request->proxy_node->interface,
178  &request->async_transaction,
179  10*1000 // Retry for thirty seconds.
180  );
181 
182 bail:
183  return realsize;
184 }

186 smcp_status_t
187 smcp_curl_proxy_request_handler(
188  smcp_curl_proxy_node_t node
189 ) {
190  smcp_status_t ret = SMCP_STATUS_NOT_ALLOWED;
191  smcp_curl_request_t request = NULL;
192  struct curl_slist *headerlist=NULL;
193  smcp_method_t method = smcp_inbound_get_code();
194 
195  //require_action(method<=COAP_METHOD_DELETE,bail,ret = SMCP_STATUS_NOT_ALLOWED);
196 
197  //require_action(COAP_OPTION_URI_PATH!=smcp_inbound_peek_option(NULL,NULL),bail,ret=SMCP_STATUS_NOT_FOUND);
198 
199  node->interface = smcp_get_current_instance();
200 
201  smcp_inbound_reset_next_option();
202 
203  request = smcp_curl_request_create();
204  request->proxy_node = node;
205 
206  require_action(request!=NULL,bail,ret = SMCP_STATUS_MALLOC_FAILURE);
207 
208  switch(method) {
209  case COAP_METHOD_GET: curl_easy_setopt(request->curl, CURLOPT_CUSTOMREQUEST, "GET"); break;
210  case COAP_METHOD_PUT: curl_easy_setopt(request->curl, CURLOPT_PUT, 1L); break;
211  case COAP_METHOD_POST: curl_easy_setopt(request->curl, CURLOPT_POST, 1L); break;
212  case COAP_METHOD_DELETE: curl_easy_setopt(request->curl, CURLOPT_CUSTOMREQUEST, "DELETE"); break;
213  default:
214  ret = SMCP_STATUS_NOT_ALLOWED;
215  break;
216  }
217 
218  {
219  coap_option_key_t key;
220  const uint8_t* value;
221  size_t value_len;
222  while((key=smcp_inbound_next_option(&value, &value_len))!=COAP_OPTION_INVALID) {
223  if(key==COAP_OPTION_PROXY_URI) {
224  char uri[value_len+1];
225  memcpy(uri,value,value_len);
226  uri[value_len] = 0;
227  curl_easy_setopt(request->curl, CURLOPT_URL, uri);
228  assert_printf("CuRL URL: \"%s\"",uri);
229  ret = 0;
230  } else if(key==COAP_OPTION_URI_HOST) {
231  } else if(key==COAP_OPTION_URI_PORT) {
232  } else if(key==COAP_OPTION_URI_PATH) {
233  } else if(key==COAP_OPTION_URI_QUERY) {
234  } else if(key==COAP_OPTION_CONTENT_TYPE || key==COAP_OPTION_ACCEPT) {
235  const char* option_name = coap_option_key_to_cstr(key, false);
236  const char* value_string = coap_content_type_to_cstr(value[1]);
237  char header[strlen(option_name)+strlen(value_string)+3];
238  strcpy(header,option_name);
239  strcat(header,": ");
240  strcat(header,value_string);
241  headerlist = curl_slist_append(headerlist, header);
242  assert_printf("CuRL HEADER: \"%s\"",header);
243  } else {
244  if(coap_option_value_is_string(key)) {
245  const char* option_name = coap_option_key_to_cstr(key, false);
246  char header[strlen(option_name)+value_len+3];
247  strcpy(header,option_name);
248  strcat(header,": ");
249  strncat(header,(const char*)value,value_len);
250  assert_printf("CuRL HEADER: \"%s\"",header);
251  headerlist = curl_slist_append(headerlist, header);
252  }
253  }
254  }
255  }
256 
257  require_noerr(ret,bail);
258 
259  if(smcp_inbound_get_content_len()) {
260  size_t len = smcp_inbound_get_content_len();
261  request->output_content = calloc(1,len+1);
262  request->output_content_len = len;
263  memcpy(request->output_content,smcp_inbound_get_content_ptr(),len);
264  curl_easy_setopt(request->curl, CURLOPT_READFUNCTION, ReadMemoryCallback);
265  curl_easy_setopt(request->curl, CURLOPT_READDATA, (void *)request);
266  }
267 
268  curl_easy_setopt(request->curl, CURLOPT_USERAGENT, "smcp-curl-proxy/1.0");
269  curl_easy_setopt(request->curl, CURLOPT_HTTPHEADER, headerlist),headerlist=NULL;
270  curl_easy_setopt(request->curl, CURLOPT_WRITEFUNCTION, WriteMemoryCallback);
271  curl_easy_setopt(request->curl, CURLOPT_WRITEDATA, (void *)request);
272 
273  ret = smcp_start_async_response(&request->async_response,0);
274  require_noerr(ret,bail);
275 
276  if(node->curl_multi_handle)
277  curl_multi_add_handle(node->curl_multi_handle, request->curl);
278  else
279  curl_easy_perform(request->curl);
280 
281 bail:
282  if(headerlist)
283  curl_slist_free_all(headerlist);
284 
285  if(ret && request)
286  smcp_curl_request_release(request);
287 
288  return ret;
289 }
