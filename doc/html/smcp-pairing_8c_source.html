503 static smcp_status_t
504 smcp_event_response_handler(
505  int statuscode, smcp_pairing_node_t pairing
506 ) {
507  DEBUG_PRINTF("Pairing:%p: smcp_event_response_handler(): statuscode=%d",pairing,statuscode);
508 
509  if(statuscode==SMCP_STATUS_TIMEOUT) {
510  if(SHOULD_CONFIRM_EVENT_FOR_PAIRING(pairing)) {
511  if(pairing->flags&SMCP_PAIRING_FLAG_OBSERVE)
512  statuscode = SMCP_STATUS_RESET;
513  } else {
514  statuscode = SMCP_STATUS_OK;
515  }
516  }
517 
518  if(statuscode!=SMCP_STATUS_TRANSACTION_INVALIDATED) {
519  if(statuscode && ((statuscode < COAP_RESULT_200) || (statuscode >= COAP_RESULT_400))) {
520  // Looks like we failed to live up to this pairing.
521 #if SMCP_CONF_PAIRING_STATS
522  pairing->errors++;
523 #endif
524  if( (pairing->flags & SMCP_PAIRING_FLAG_OBSERVE)
525 #if SMCP_CONF_PAIRING_STATS
526  && (pairing->errors>=10)
527 #endif
528  ) {
529  DEBUG_PRINTF("Event:%p: Too many errors!\n",pairing);
530  statuscode = SMCP_STATUS_RESET;
531  }
532  if(statuscode==SMCP_STATUS_RESET && (pairing->flags&SMCP_PAIRING_FLAG_OBSERVE)) {
533  smcp_delete_pairing(pairing);
534  return SMCP_STATUS_OK;
535  }
536 #if SMCP_CONF_PAIRING_STATS
537  smcp_trigger_event_with_node(
538  smcp_get_current_instance(),
539  &pairing->node.node,
540  "ec"
541  );
542 #endif
543  }
544 #if SMCP_CONF_PAIRING_STATS
545  if(pairing->last_error != statuscode) {
546  pairing->last_error = statuscode;
547  smcp_trigger_event_with_node(
548  smcp_get_current_instance(),
549  &pairing->node.node,
550  "err"
551  );
552  }
553 #endif
554  }
555 
556 #if SMCP_CONF_USE_SEQ
557  if(!statuscode || ((statuscode >= COAP_RESULT_200) && (statuscode < COAP_RESULT_400)))
558  pairing->ack = pairing->seq;
559 #endif
560 
561  // expire the event.
562  {
563  smcp_event_tracker_t event = pairing->currentEvent;
564  pairing->currentEvent = NULL;
565  smcp_event_tracker_release(event);
566  }
567  return SMCP_STATUS_OK;
568 }

752 
753  if(!path) path = "";
754 
755  // Move past any preceding slashes.
756  while(path && path[0] == '/' && path[1] == 0) path++;
757 
758  for(iter = smcp_get_first_pairing_for_path(self, path);

773

776  event = calloc(1,sizeof(*event));

793  (void*)&smcp_event_response_handler,
794  (void*)iter
795  );

812  iter->fire_count++;
813 #endif
814  }
