83 smcp_status_t
84 smcp_node_route(smcp_node_t node, smcp_request_handler_func* func, void** context) {
85  smcp_status_t ret = 0;
86  smcp_t const self = smcp_get_current_instance();
87 
88  smcp_inbound_reset_next_option();
89 
90  {
91  // TODO: Rewrite this to be more efficient.
92  const uint8_t* prev_option_ptr = self->inbound.this_option;
93  coap_option_key_t prev_key = 0;
94  coap_option_key_t key;
95  const uint8_t* value;
96  size_t value_len;
97  while((key=smcp_inbound_next_option(&value, &value_len))!=COAP_OPTION_INVALID) {
98  if(key>COAP_OPTION_URI_PATH) {
99  self->inbound.this_option = prev_option_ptr;
100  self->inbound.last_option_key = prev_key;
101  break;
102  } else if(key==COAP_OPTION_URI_PATH) {
103  smcp_node_t next = smcp_node_find(
104  node,
105  (const char*)value,
106  (int)value_len
107  );
108  if(next) {
109  node = next;
110  } else {
111  self->inbound.this_option = prev_option_ptr;
112  self->inbound.last_option_key = prev_key;
113  break;
114  }
115  } else if(key==COAP_OPTION_URI_HOST) {
116  // Skip host at the moment,
117  // because we don't do virtual hosting yet.
118  } else if(key==COAP_OPTION_URI_PORT) {
119  // Skip port at the moment,
120  // because we don't do virtual hosting yet.
121  } else if(key==COAP_OPTION_PROXY_URI) {
122  // Skip the proxy URI for now.
123  } else if(key==COAP_OPTION_CONTENT_TYPE) {
124  // Skip.
125  } else {
126  if(COAP_OPTION_IS_CRITICAL(key)) {
127  ret=SMCP_STATUS_BAD_OPTION;
128  assert_printf("Unrecognized option %d, \"%s\"",
129  key,
130  coap_option_key_to_cstr(key, false)
131  );
132  goto bail;
133  }
134  }
135  prev_option_ptr = self->inbound.this_option;
136  prev_key = self->inbound.last_option_key;
137  }
138  }
139 
140  *func = (void*)node->request_handler;
141  if(node->context)
142  *context = node->context;
143  else
144  *context = (void*)node;
145 
146 bail:
147  return ret;
148 }

332 int
333 smcp_node_find_next_with_path(
334  smcp_node_t node,
335  const char* orig_path, // Escaped.
336  smcp_node_t* next
337 ) {
338  const char* path = orig_path;
339 
340  require(next, bail);
341  require(node, bail);
342  require(path, bail);
343 
344  // Move past any preceding slashes.
345  while(path[0] == '/')
346  path++;
347 
348  if(path[0] == 0) {
349  // Self.
350  *next = node;
351  } else {
352  // Device or Variable.
353  int namelen;
354  for(namelen = 0; path[namelen]; namelen++) {
355  if((path[namelen] == '/') || (path[namelen] == '?') ||
356  (path[namelen] == '!'))
357  break;
358  }
359 
360  // Warning: This could be dangerous!
361  // We should evaluate the liklihood of blowing
362  // the stack here. TODO: Investigate potential oveflow!
363  {
364 #if HAVE_C99_VLA
365  char unescaped_name[namelen+1];
366 #else
367  char *unescaped_name = malloc(namelen+1);
368 #endif
369  size_t escaped_len = url_decode_str(
370  unescaped_name,
371  namelen+1,
372  path,
373  namelen
374  );
375  *next = smcp_node_find(
376  node,
377  unescaped_name,
378  (int)escaped_len
379  );
380 #if !HAVE_C99_VLA // && !SMCP_AVOID_MALLOC
381  free(unescaped_name);
382 #endif
383  }
384  if(!*next) {
385  DEBUG_PRINTF(
386  "Unable to find node. node->name=%s, path=%s, namelen=%d",
387  node->name, path, namelen);
388  goto bail;
389  }
390  }
391 
392  if(!*next) {
393  DEBUG_PRINTF(
394  "Unable to find node. node->name=%s, path=%s", node->name,
395  path);
396  goto bail;
397  }
398 
399  // Move to next name
400  while(path[0] && (path[0] != '/') && (path[0] != '!') &&
401  (path[0] != '?'))
402  path++;
403 
404  // Move past any preceding slashes.
405  while(path[0] == '/')
406  path++;
407 
408 bail:
409  return (int)(path - orig_path);
410 }
