89 smcp_status_t
90 smcp_outbound_begin(
91  smcp_t self, coap_code_t code, coap_transaction_type_t tt
92 ) {
93  SMCP_EMBEDDED_SELF_HOOK;
94 #if !SMCP_EMBEDDED
95  if(!self)
96  self = smcp_get_current_instance();
97 #endif
98 
99  check(!smcp_get_current_instance() || smcp_get_current_instance()==self);
100  smcp_set_current_instance(self);
101 
102 #if SMCP_USE_BSD_SOCKETS
103  self->outbound.packet = (struct coap_header_s*)self->outbound.packet_bytes;
104 #elif CONTIKI
105  uip_udp_conn = self->udp_conn;
106  self->outbound.packet = (struct coap_header_s*)&uip_buf[UIP_LLH_LEN + UIP_IPUDPH_LEN];
107  if(self->outbound.packet==self->inbound.packet) {
108  self->outbound.packet = (struct coap_header_s*)&uip_buf[UIP_LLH_LEN + UIP_IPUDPH_LEN + self->inbound.packet_len + 1];
109  }
110 #else
111 #warning WRITEME!
112 #endif
113 
114  self->outbound.packet->tt = tt;
115  self->outbound.packet->msg_id = self->outbound.next_tid;
116  self->outbound.packet->code = code;
117  self->outbound.packet->version = COAP_VERSION;
118 
119  // Set the token.
120  if( self->is_processing_message
121  && self->inbound.packet->token_len
122  && code
123  && tt != COAP_TRANS_TYPE_RESET
124  ) {
125  self->outbound.packet->token_len = self->inbound.packet->token_len;
126  memcpy(self->outbound.packet->token,self->inbound.packet->token,self->outbound.packet->token_len);
127  } else if(code && code<COAP_RESULT_100 && self->current_transaction) {
128  // For sending a request.
129  self->outbound.packet->token_len = sizeof(self->current_transaction->token);
130  memcpy(self->outbound.packet->token,(void*)&self->current_transaction->token,self->outbound.packet->token_len);
131  } else {
132  self->outbound.packet->token_len = 0;
133  }
134 
135  self->outbound.last_option_key = 0;
136 
137  self->outbound.content_ptr = (char*)self->outbound.packet->token + self->outbound.packet->token_len;
138  *self->outbound.content_ptr++ = 0xFF; // Add end-of-options marker
139  self->outbound.content_len = 0;
140  self->force_current_outbound_code = false;
141  self->is_responding = false;
142 
143 #if SMCP_USE_BSD_SOCKETS
144  self->outbound.socklen = 0;
145 #elif defined(CONTIKI)
146  // Writeme!
147 #endif
148  return SMCP_STATUS_OK;
149 }

151 smcp_status_t smcp_outbound_begin_response(coap_code_t code) {
152  smcp_status_t ret = SMCP_STATUS_OK;
153  smcp_t const self = smcp_get_current_instance();
154 
155  require_action_string(!self->did_respond,
156  bail,
157  ret = SMCP_STATUS_RESPONSE_NOT_ALLOWED,
158  "Attempted to send more than one response!"
159  );
160 
161  // If we have already started responding, don't bother.
162  require(!self->is_responding,bail);
163 
164  if(self->is_processing_message)
165  self->outbound.next_tid = smcp_inbound_get_msg_id();
166  smcp_outbound_begin(self,code,(self->inbound.packet->tt==COAP_TRANS_TYPE_NONCONFIRMABLE)?COAP_TRANS_TYPE_NONCONFIRMABLE:COAP_TRANS_TYPE_ACK);
167  self->is_responding = true;
168 
169  if(self->is_processing_message)
170  smcp_outbound_set_msg_id(smcp_inbound_get_msg_id());
171 
172  if(self->is_processing_message) {
173 #if SMCP_USE_BSD_SOCKETS
174  ret = smcp_outbound_set_destaddr(
175  self->inbound.saddr,
176  self->inbound.socklen
177  );
178 #elif CONTIKI
179  ret = smcp_outbound_set_destaddr(
180  &self->inbound.toaddr,
181  self->inbound.toport
182  );
183 #endif
184  require_noerr(ret, bail);
185  }
186 bail:
187  return ret;
188 }

257 static smcp_status_t
258 smcp_outbound_add_option_(
259  coap_option_key_t key, const char* value, size_t len
260 ) {
261  smcp_t const self = smcp_get_current_instance();
262 
263  if( ( self->outbound.content_ptr
264  - (char*)self->outbound.packet
265  + len + 10
266  ) > SMCP_MAX_PACKET_LENGTH
267  ) {
268  // We ran out of room!
269  return SMCP_STATUS_MESSAGE_TOO_BIG;
270  }
271 
272  if(key<self->outbound.last_option_key) {
273  assert_printf("warning: Out of order header: %s",coap_option_key_to_cstr(key, self->is_responding));
274  }
275 
276  if(len == SMCP_CSTR_LEN)
277  len = strlen(value);
278 
279  if(self->outbound.content_ptr!=(char*)self->outbound.packet->token+self->outbound.packet->token_len)
280  self->outbound.content_ptr--; // remove end-of-options marker
281 
282  self->outbound.content_ptr += coap_insert_option(
283  (uint8_t*)self->outbound.packet->token+self->outbound.packet->token_len,
284  (uint8_t*)self->outbound.content_ptr,
285  key,
286  (const uint8_t*)value,
287  len
288  );
289 
290  if(key>self->outbound.last_option_key)
291  self->outbound.last_option_key = key;
292 
293  *self->outbound.content_ptr++ = 0xFF; // Add end-of-options marker
294 
295 #if OPTION_DEBUG
296  coap_dump_header(
297  SMCP_DEBUG_OUT_FILE,
298  "Option-Debug >>> ",
299  self->outbound.packet,
300  self->outbound.content_ptr-(char*)self->outbound.packet
301  );
302 #endif
303 
304  return SMCP_STATUS_OK;
305 }

307 static smcp_status_t
308 smcp_outbound_add_options_up_to_key_(
309  coap_option_key_t key
310 ) {
311  smcp_status_t ret = SMCP_STATUS_OK;
312  smcp_t const self = smcp_get_current_instance();
313 
314 #if SMCP_CONF_TRANS_ENABLE_BLOCK2
315  if( (self->current_transaction
316  && self->current_transaction->next_block2)
317  && self->outbound.last_option_key<COAP_OPTION_BLOCK2
318  && key>COAP_OPTION_BLOCK2
319  ) {
320  uint32_t block2 = htonl(self->current_transaction->next_block2);
321  uint8_t size = 3; // TODO: calculate this properly
322  ret = smcp_outbound_add_option_(
323  COAP_OPTION_BLOCK2,
324  (char*)&block2+sizeof(block2)-size,
325  size
326  );
327  }
328 #endif
329 
330 #if SMCP_CONF_TRANS_ENABLE_OBSERVING
331  if( (self->current_transaction && self->current_transaction->flags&SMCP_TRANSACTION_OBSERVE)
332  && self->outbound.last_option_key<COAP_OPTION_OBSERVE
333  && key>COAP_OPTION_OBSERVE
334  ) {
335  if(self->outbound.packet->code && self->outbound.packet->code<COAP_RESULT_100) {
336  // For sending a request.
337  ret = smcp_outbound_add_option_(
338  COAP_OPTION_OBSERVE,
339  (void*)NULL,
340  0
341  );
342  }
343  }
344 #endif
345 
346  if( self->outbound.last_option_key<COAP_OPTION_AUTHENTICATE
347  && key>COAP_OPTION_AUTHENTICATE
348  ) {
349  ret = smcp_auth_add_options();
350  }
351 
352 #if SMCP_USE_CASCADE_COUNT
353  if( self->outbound.last_option_key<COAP_OPTION_CASCADE_COUNT
354  && key>COAP_OPTION_CASCADE_COUNT
355  && self->has_cascade_count
356  ) {
357  ret = smcp_outbound_add_option_(
358  COAP_OPTION_CASCADE_COUNT,
359  (char*)&self->cascade_count,
360  1
361  );
362  }
363 #endif
364 
365  return ret;
366 }

407 smcp_status_t
408 smcp_outbound_set_destaddr_from_host_and_port(const char* addr_str,uint16_t toport) {
409  smcp_status_t ret = SMCP_STATUS_OK;
410 
411 #if SMCP_USE_BSD_SOCKETS
412  struct sockaddr_in6 saddr = {
413  .sin6_family = AF_INET6,
414 #if SOCKADDR_HAS_LENGTH_FIELD
415  .sin6_len = sizeof(struct sockaddr_in6),
416 #endif
417  };
418 
419  struct addrinfo hint = {
420  .ai_flags = AI_ADDRCONFIG,
421  .ai_family = AF_UNSPEC,
422  };
423 
424  struct addrinfo *results = NULL;
425  struct addrinfo *iter = NULL;
426 
427  // Check to see if this host is a group we know about.
428  if(strcasecmp(addr_str, "coap-alldevices")==0)
429  addr_str = COAP_MULTICAST_IP6_ALLDEVICES;
430 
431  int error = getaddrinfo(addr_str, NULL, &hint, &results);
432 
433  DEBUG_PRINTF("Outbound: Dest host [%s]:%d",addr_str,toport);
434  toport = htons(toport);
435 
436  if(error && (inet_addr(addr_str) != INADDR_NONE)) {
437  char addr_v4mapped_str[8 + strlen(addr_str)];
438  hint.ai_family = AF_INET6;
439  hint.ai_flags = AI_ALL | AI_V4MAPPED,
440  strcpy(addr_v4mapped_str,"::ffff:");
441  strcat(addr_v4mapped_str,addr_str);
442  error = getaddrinfo(addr_v4mapped_str,
443  NULL,
444  &hint,
445  &results
446  );
447  }
448 
449  require_action(error!=EAI_AGAIN,bail,ret=SMCP_STATUS_WAIT_FOR_DNS);
450 
451 #ifdef TM_EWOULDBLOCK
452  require_action(error!=TM_EWOULDBLOCK,bail,ret=SMCP_STATUS_WAIT_FOR_DNS);
453 #endif
454 
455  require_action_string(
456  !error,
457  bail,
458  ret = SMCP_STATUS_HOST_LOOKUP_FAILURE,
459  gai_strerror(error)
460  );
461 
462  for(iter = results;iter && (iter->ai_family!=AF_INET6 && iter->ai_family!=AF_INET);iter=iter->ai_next);
463 
464  require_action(
465  iter,
466  bail,
467  ret = SMCP_STATUS_HOST_LOOKUP_FAILURE
468  );
469 
470  if(iter->ai_family == AF_INET) {
471  struct sockaddr_in *v4addr = (void*)iter->ai_addr;
472  saddr.sin6_addr.s6_addr[10] = 0xFF;
473  saddr.sin6_addr.s6_addr[11] = 0xFF;
474  memcpy(&saddr.sin6_addr.s6_addr[12], &v4addr->sin_addr.s_addr, 4);
475  } else {
476  if(saddr.sin6_addr.s6_addr[0]==0xFF) {
477  smcp_t const self = smcp_get_current_instance();
478  check(self->outbound.packet->tt != COAP_TRANS_TYPE_CONFIRMABLE);
479  if(self->outbound.packet->tt == COAP_TRANS_TYPE_CONFIRMABLE) {
480  self->outbound.packet->tt = COAP_TRANS_TYPE_NONCONFIRMABLE;
481  }
482  }
483  memcpy(&saddr, iter->ai_addr, iter->ai_addrlen);
484  }
485  saddr.sin6_port = toport;
486 
487  ret = smcp_outbound_set_destaddr((struct sockaddr *)&saddr,sizeof(struct sockaddr_in6));
488  require_noerr(ret, bail);
489 
490 #elif CONTIKI
491  uip_ipaddr_t toaddr;
492  memset(&toaddr, 0, sizeof(toaddr));
493 
494  DEBUG_PRINTF("Outbound: Dest host [%s]:%d",addr_str,toport);
495  toport = htons(toport);
496 
497  ret = uiplib_ipaddrconv(
498  addr_str,
499  &toaddr
500  ) ? SMCP_STATUS_OK : SMCP_STATUS_HOST_LOOKUP_FAILURE;
501 
502 #if SMCP_CONF_USE_DNS
503  if(ret) {
504  uip_ipaddr_t *temp = NULL;
505 #if RESOLV_SUPPORTS_LOOKUP2_API
506  switch(resolv_lookup2(addr_str,&temp)) {
507  case RESOLV_STATUS_CACHED:
508  memcpy(&toaddr, temp, sizeof(uip_ipaddr_t));
509  ret = SMCP_STATUS_OK;
510  break;
511  case RESOLV_STATUS_UNCACHED:
512  case RESOLV_STATUS_EXPIRED:
513  resolv_query(addr_str);
514  case RESOLV_STATUS_RESOLVING:
515  ret = SMCP_STATUS_WAIT_FOR_DNS;
516  break;
517  default:
518  case RESOLV_STATUS_ERROR:
519  case RESOLV_STATUS_NOT_FOUND:
520  ret = SMCP_STATUS_HOST_LOOKUP_FAILURE;
521  break;
522  }
523 #else // RESOLV_SUPPORTS_LOOKUP2_API
524  // Fall back for older versions of contiki
525  // which don't have the more advanced DNS resolver API.
526  temp = resolv_lookup(addr_str);
527  if(temp) {
528  memcpy(&toaddr, temp, sizeof(uip_ipaddr_t));
529  ret = SMCP_STATUS_OK;
530  } else {
531  resolv_query(addr_str);
532  ret = SMCP_STATUS_WAIT_FOR_DNS;
533  }
534 #endif // !RESOLV_SUPPORTS_LOOKUP2_API
535  }
536 #endif // SMCP_CONF_USE_DNS
537 
538  require_noerr(ret,bail);
539 
540  ret = smcp_outbound_set_destaddr(&toaddr,toport);
541 
542  require_noerr(ret, bail);
543 #endif // CONTIKI
544 
545 bail:
546 #if SMCP_USE_BSD_SOCKETS
547  if(results)
548  freeaddrinfo(results);
549 #endif
550  return ret;
551 }

553 smcp_status_t
554 smcp_outbound_set_uri(
555  const char* uri, char flags
556 ) {
557  smcp_status_t ret = SMCP_STATUS_OK;
558  smcp_t const self = smcp_get_current_instance();
559  char* proto_str = NULL;
560  char* addr_str = NULL;
561  char* port_str = NULL;
562  char* path_str = NULL;
563  char* query_str = NULL;
564  char* uri_copy = NULL;
565  char* username_str = NULL;
566  char* password_str = NULL;
567  uint16_t toport = SMCP_DEFAULT_PORT;
568 
569  require_action(uri, bail, ret = SMCP_STATUS_INVALID_ARGUMENT);
570 
571  {
572 #if HAVE_ALLOCA
573  uri_copy = alloca(strlen(uri) + 1);
574  strcpy(uri_copy, uri);
575 #else
576  uri_copy = strdup(uri);
577 #endif
578 
579  require_action(uri_copy!=NULL,bail,ret = SMCP_STATUS_MALLOC_FAILURE);
580 
581  // Parse the URI.
582  require_action_string(
583  url_parse(
584  uri_copy,
585  &proto_str,
586  &username_str,
587  &password_str,
588  &addr_str,
589  &port_str,
590  &path_str,
591  &query_str
592  ),
593  bail,
594  ret = SMCP_STATUS_URI_PARSE_FAILURE,
595  "Unable to parse URL"
596  );
597 
598  if(!proto_str && !addr_str) {
599  // Talking to ourself.
600  proto_str = "coap";
601  addr_str = "::1";
602  toport = smcp_get_port(smcp_get_current_instance());
603  flags |= SMCP_MSG_SKIP_AUTHORITY;
604  } else if(port_str) {
605  toport = atoi(port_str);
606  }
607 
608  DEBUG_PRINTF(
609  CSTR("URI Parse: \"%s\" -> host=\"%s\" port=\"%u\" path=\"%s\""),
610  uri,
611  addr_str,
612  toport,
613  path_str
614  );
615  }
616 
617  if( proto_str && !strequal_const(proto_str, "coap") ) {
618  require_action_string(
619  self->proxy_url,
620  bail,
621  ret=SMCP_STATUS_INVALID_ARGUMENT,
622  "No proxy URL configured"
623  );
624  require_action(uri!=self->proxy_url,bail,ret = SMCP_STATUS_INVALID_ARGUMENT);
625 
626  ret = smcp_outbound_add_option(COAP_OPTION_PROXY_URI, uri, strlen(uri));
627  require_noerr(ret, bail);
628  ret = smcp_outbound_set_uri(self->proxy_url,flags);
629  goto bail;
630  }
631 
632  if(!(flags&SMCP_MSG_SKIP_AUTHORITY)) {
633  if(addr_str && !string_contains_colons(addr_str)) {
634  ret = smcp_outbound_add_option(COAP_OPTION_URI_HOST, addr_str, strlen(addr_str));
635  require_noerr(ret, bail);
636  }
